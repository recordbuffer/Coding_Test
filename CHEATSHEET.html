빈도 세기 · Counter

개수 세기
from collections import Counter
c = Counter(arr)      # {값:개수}
c["a"]                # 없으면 0 (에러X)
빼기 (없는 것만 남기기)
Counter(a) - Counter(b)
# 완주 못한 선수: 참가-완주
list(Counter(a)-Counter(b))[0]
가장 많은 것 N개
c.most_common(2)   # [(값,개수),...]

정렬 · sorted / key

기본
sorted(arr)              # 새 리스트
arr.sort()               # 원본 변경, 반환 None
sorted(arr, reverse=True)
기준 정하기 (key)
sorted(arr, key=lambda x: x[1])
# 각 원소의 x[1]로 정렬
다단계: 1순위, 2순위
sorted(arr, key=lambda x: (x[0], x[1]))
# x[0]로, 같으면 x[1]로
혼합 방향 (숫자만 내림)
key=lambda x: (-x[1], x[0])
# 점수 내림, 같으면 이름 오름
※ 음수는 숫자만. reverse는 전체 뒤집음
핵심 3
1) key = "무엇을 기준으로"
2) 튜플 = "1순위, 2순위" # 앞 같을때 뒤
3) 숫자 내림 = 음수(-)

스택 · 큐 · deque

스택 (뒤로 넣고 뒤로 빼기)
stack = []
stack.append(x)
stack.pop()        # 마지막 꺼냄
stack[-1]          # 안 빼고 보기(top)
큐 (앞으로 빼기) = deque
from collections import deque
q = deque()
q.append(x)        # 뒤에 넣기
q.popleft()        # 앞에서 빼기 O(1)
※ list.pop(0)는 느림 → 큐는 꼭 deque
짝맞추기 패턴 (괄호검사)
for c in s:
    if c == "(":
        stack.append(c)
    else:                  # 닫는 괄호
        if not stack: return False
        stack.pop()
return not stack   # 남으면 실패
