SELECT / 정렬  (재풀이 2026-07-21, MySQL)

1. 모든 레코드 조회하기
동물 보호소에 들어온 모든 동물의 정보를 ANIMAL_ID순으로 조회.
SELECT * FROM ANIMAL_INS
ORDER BY ANIMAL_ID;


2. 역순 정렬하기
모든 동물의 이름과 보호 시작일을 ANIMAL_ID 역순으로 조회.
SELECT NAME, DATETIME FROM ANIMAL_INS
ORDER BY ANIMAL_ID DESC;


3. 아픈 동물 찾기
아픈 동물(INTAKE_CONDITION='Sick')의 아이디와 이름을 아이디 순으로 조회.
SELECT ANIMAL_ID, NAME FROM ANIMAL_INS
WHERE INTAKE_CONDITION = 'Sick'
ORDER BY ANIMAL_ID;


4. 어린 동물 찾기
젊은 동물(Aged 아님)의 아이디와 이름을 아이디 순으로 조회.
SELECT ANIMAL_ID, NAME FROM ANIMAL_INS
WHERE INTAKE_CONDITION != 'Aged'
ORDER BY ANIMAL_ID;
-- 주의: != 는 NULL 행을 빼버린다. 이 테이블엔 NULL이 없어 통과.
--       "Aged 아닌 것 전부"가 목적이면 OR INTAKE_CONDITION IS NULL 필요.


5. 동물의 아이디와 이름
모든 동물의 아이디와 이름을 ANIMAL_ID순으로 조회.
SELECT ANIMAL_ID, NAME FROM ANIMAL_INS
ORDER BY ANIMAL_ID;


6. 여러 기준으로 정렬하기
모든 동물의 아이디, 이름, 보호 시작일을 이름순으로. 이름이 같으면 나중에 시작한 동물 먼저.
SELECT ANIMAL_ID, NAME, DATETIME FROM ANIMAL_INS
ORDER BY NAME, DATETIME DESC;
-- 처음에 WHERE NAME IS NOT NULL 을 붙여 실패함.
-- "모든 동물"이라 NULL도 포함해야 한다 → 필터 쓰면 안 됨.


7. 상위 n개 레코드
가장 먼저 들어온 동물의 이름을 조회.
SELECT NAME FROM ANIMAL_INS
ORDER BY DATETIME LIMIT 1;
